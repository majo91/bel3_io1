59 void _initUART0_CH0() {
60  /* USIC channels initialization */
61  XMC_UART_CH_Init(XMC_UART0_CH0, &uart_config);
62 
63  XMC_UART_CH_SetInputSource(XMC_UART0_CH0, XMC_UART_CH_INPUT_RXD, USIC0_C0_DX0_P1_4);
64 
65  /* FIFOs initialization for both channels:
66  * 8 entries for TxFIFO from point 0, LIMIT=1
67  * 8 entries for RxFIFO from point 8, LIMIT=7 (SRBI is set if all 8*data has been received)
68  * */
69  XMC_USIC_CH_TXFIFO_Configure(XMC_UART0_CH0, 0, XMC_USIC_CH_FIFO_SIZE_8WORDS, TX_FIFO_INITIAL_LIMIT);
70  XMC_USIC_CH_RXFIFO_Configure(XMC_UART0_CH0, 8, XMC_USIC_CH_FIFO_SIZE_8WORDS, RX_FIFO_INITIAL_LIMIT);
71 
72  /* Enabling events for TX FIFO and RX FIFO */
73  XMC_USIC_CH_RXFIFO_EnableEvent(XMC_UART0_CH0,XMC_USIC_CH_RXFIFO_EVENT_CONF_STANDARD |
74  XMC_USIC_CH_RXFIFO_EVENT_CONF_ALTERNATE);
75 
76  /* Connecting the previously enabled events to a Service Request line number */
77  XMC_USIC_CH_RXFIFO_SetInterruptNodePointer(XMC_UART0_CH0,XMC_USIC_CH_RXFIFO_INTERRUPT_NODE_POINTER_STANDARD,0);
78  XMC_USIC_CH_RXFIFO_SetInterruptNodePointer(XMC_UART0_CH0,XMC_USIC_CH_RXFIFO_INTERRUPT_NODE_POINTER_ALTERNATE,0);
